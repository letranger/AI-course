1. Tensorflow use GPU

 1: import tensorflow as tf
 2: import time
 3: def time_matmul(x):
 4:     start = time.time()
 5:     for loop in range(10):
 6:         tf.matmul(x, x)
 7:         result = time.time()-start
 8:         print("10 loops: {:0.2f}ms".format(1000*result))
 9: 
10: # Force execution on CPU print("On CPU:")
11: with tf.device("CPU:0"):
12:     print("On CPU:")
13:     x = tf.random.uniform([1000, 1000])
14:     assert x.device.endswith("CPU:0")
15:     time_matmul(x)
16: 
17: # Force execution on GPU #0 if available
18: if tf.test.is_gpu_available():
19:     print("On GPU:")
20:     with tf.device("GPU:0"): # Or GPU:1 for the 2nd GPU, GPU:2 for the 3rd etc.
21:         x = tf.random.uniform([1000, 1000])
22:         assert x.device.endswith("GPU:0")
23:         time_matmul(x)
